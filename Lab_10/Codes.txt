int main() {
    int n;
    cin >> n;
    vector<string> names;
    vector<int> marks;
    for(int i=0;i<n;i++){
        string s;
        int c;
        cin>>s>>c;
        names.push_back(s);
        marks.push_back(c);
    }
    sort(names.begin(), names.end());
    sort(marks.begin(), marks.end());

    for (int i = 0; i < n; i++)
    {
        cout << names[i] << " " << marks[i] << endl;
    }

    return 0;
}

#include <bits/stdc++.h>
using namespace std;
long long fact (int v){
  long long z=1;
  for (long i=1;i<=v;i++){
    z*=v;
  }
  return z;
}
int main(){
    int n;
    cin>>n;
    vector<long long> s;
    for(int i=0; i<=n; i++){
        s.push_back(fact(i));
    }
    for(int i=0; i<=n; i++){
        cout<<s[i]<<" ";
    }
}

bool f(vector<int> t,vector<int> v){
    int a=0,b=0;

        a+=t[i];
    }
